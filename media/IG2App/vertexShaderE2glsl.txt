#version 330 core

in vec4 vertex; // atributos de los vértices a procesar

in vec2 uv0;

uniform mat4 modelViewProjMat; // constantes de programa

out vec2 vUv0; // out del vertex shader

out vec2 s;

void main() {
    s.x += 0.5;
    s.y += 0.5;

    vUv0 = uv0;
    gl_Position = modelViewProjMat * vertex; // obligatorio
}
